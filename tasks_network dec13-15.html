Tasks at the same time - Facebook Hacker Contest¶

The point of this notebook is to show a solution that can be better visualized and utilizes my favorite python package, networkx. This way, every task is a node of a graph and two tasks are connected by an edge in case they overlap. Our job is to remove all overlaps (every edge) by removing the least possible nodes and the idea for that is to remove the nodes with the highest degree first, until the whole network is disconnected. Afterwards we can count the remaining tasks and show which tasks are these.

def make_2input_lists(n_task):
    startlist = [rng.randint(1,20) for i in range(n_task)]
    #we just sort it so the tasks are more readable, end list should remain the same
    startlist.sort() 
    endlist = [start+rng.randint(0,5) for start in startlist]

    print(f"Created end list: {endlist}")
    
    return startlist, endlist

#make a graph from lists
def create_graph(start_list, end_list):
    G = nx.Graph() #initialize graph
    n=len(start_list) #number of tasks
    
    #add as many deg=0 nodes as many tasks
    for node in range(n):
        G.add_node(node)
        
    #create two info holding list which we can just modify
    node_list = [i for i in range(n)]
    start_list_copy = start_list.copy()
    
    #add edges as needed
    for task in range(n-1):
        next_task_index = start_list_copy.index(min(start_list_copy))
        next_task_end = end_list[next_task_index]
        
        #remove them from the info keeping lists
        node_list.remove(next_task_index)
        start_list_copy[next_task_index] = 1000 #set used task start to a 'ridiculously' large number
        
        for comp in node_list: #node_list has indices
            if start_list[comp] <= next_task_end:
                G.add_edge(comp, next_task_index)
    
    #show the graph
    nx.draw_kamada_kawai(G, with_labels=True)
    
    return G

With a while loop a node is removed at a time until the whole network is disconnected, this property is checked at the end of every iteration. The nx.density function is used to check the goal condition, by using that the density of a completely disconnected graph is 0. nx.degree is a convenient function for this task as it outputs the degree of every node of the network in dictionary, where I can look for the highest degree node, remove it from the network then update the degree dictionary in the next iteration.
This function also shows the state of the graph after each node removal for a more comprehensive look on the solving process.

def solving_graph(G):
    
    f = plt.figure(figsize=(15,10))
    f.add_subplot(3,3,1)
    #grid(True)
    nx.draw_kamada_kawai(G, with_labels=True)
    #nx.degree function is perfect here
    degrees = dict(nx.degree(G))
    
    #with a while loop can remove highest degree node until disconnect
    connected = nx.density(G) #if there are no edges density is 0
    next_im = 2
    
    while connected != 0:
        next_highest_index = max(degrees, key=degrees.get)
        
        #remove from graph and from nodelist
        del(degrees[next_highest_index])
        G.remove_node(next_highest_index)
        degrees = dict(nx.degree(G)) #need to update the degrees dict too :D #final bug
        
        #opt: print each iteration step
        f.add_subplot(3,3,next_im)
        nx.draw_kamada_kawai(G, with_labels=True)
        #grid(True)
        next_im += 1
        
        #connection condition
        connected = nx.density(G)
    
    plt.show()
         
    #print resulting task number and tasks
    remaining_node = list(G.nodes())
    task_n = len(remaining_node)
    print(f"The number of tasks which can be done by a single person is {task_n} and these are task {remaining_node}")
    
    return None
