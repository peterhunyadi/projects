SHPATH¶

https://www.spoj.com/problems/SHPATH/
You are given a list of cities. Each direct connection between two cities has its transportation cost (an integer bigger than 0). The goal is to find the paths of minimum cost between pairs of cities. Assume that the cost of each path (which is the sum of costs of all direct connections belonging to this path) is at most 200000. The name of a city is a string containing characters a, ..., z and is at most 10 characters long.

First I just visualize such a system, but I want to broaden the scope to be able to generate larger networks easily and be able to show them.

Networkx has various generators that allow the creation of random networks of different types.
https://networkx.org/documentation/stable/reference/generators.html
Here, I create a (usually) well connected Erds-Rnyi graph with a certain seed, then use Dijkstra's algorithm to calculate a shortest path between 2 nodes. If we use the same seed, the shortest path will be displayable between two different nodes afterwards.

def random_erdos_with_labels(n_node, prob, seed):
    H = nx.erdos_renyi_graph(n_node, prob, seed)
    
    for u, v in H.edges():
        H.add_edge(u, v, label = np.random.randint(9)+1)
    
    plt.figure(figsize=(16,9))
    #pos = nx.circular_layout(H)
    pos = nx.kamada_kawai_layout(H)
    nx.draw_networkx(H, pos = pos, with_label = True, node_size=600,font_size=15, #edge_color=colors, 
                     node_color='y', width=2)
    nx.draw_networkx_edge_labels(H, pos)
    
    return H

def dijkstra_sequence(source, H):

    n_node = len(H.nodes())
    dist = {}
    prev = {}
    Q = []
    
    for v in range(n_node):
        dist.update({v: np.inf})
        prev.update({v: "null"})
        Q.append(v)
    dist[source] = 0
    
    #loop
    while len(Q) != 0:
        u = np.NAN
        dist_here = np.inf
        for i in Q:
            if dist[i] < dist_here:
                dist_here = dist[i]
                u = i
        Q.remove(u)
        
        direct_neigh = [n for n in H.neighbors(u)]
        for v in direct_neigh:
            if v in Q:
                alt = dist[u] + H.get_edge_data(u, v)["label"]
                if alt < dist[v] and dist[u] is not np.inf:
                    dist[v] = alt
                    prev[v] = u
    
    return dist, prev

def sequence_source_to_target(source, target, dist, prev):

    S = []
    u = target

    if prev[u] in range(10) or u == source:          # Do something only if the vertex is reachable
        while u in range(10):                       # Construct the shortest path with a stack S
            S.insert(0, u)        # Push the vertex onto the stack
            u = prev[u]

    return S

def print_the_way(S, H):
    
    red_edges = []
    for i in range(len(S)-1):
        vertices = [S[i], S[i+1]]
        red_edges.append((min(vertices), max(vertices)))
    
    for u, v in H.edges():
        if (u,v) in red_edges:
            H.add_edge(u, v, color = 'r')
        else:
            H.add_edge(u, v, color = 'black')
    
    plt.figure(figsize=(16,9))
    #pos = nx.circular_layout(H)
    pos = nx.kamada_kawai_layout(H)
    colors = [H[u][v]['color'] for u,v in H.edges()]
    nx.draw_networkx(H, pos = pos, with_label = True, node_size=600,font_size=15, edge_color=colors, 
                     node_color='y', width=2)
    #nx.draw_networkx_edge_labels(H, pos)
    
    return H

def wrapper(source, target, n_node, prob, seed):
    H = random_erdos_with_labels(n_node, prob, seed) #generate and draw random graph according to the erdos-renyi model
    dist, prev = dijkstra_sequence(source, H) #dijkstra gets the fastest path to any node from the source
    S = sequence_source_to_target(source, target, dist, prev) #trace back from target to source
    print_the_way(S, H) #print resulting path highlighted
