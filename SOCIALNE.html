def parse_matrix(matrix):
    #as the matrix is symmetrical we can use only its upper half over the diagonal
    #add an edge to the graph for each Y sign
    G = nx.Graph()

    for m in range(len(matrix[:,0])):
        for n in range(len(matrix[0,:])):
            if m < n: #only look at them once 
                if matrix[m,n] == "Y":
                    G.add_edge(m,n)

    print(f"Number of edges: {len(G.edges())}")

    return(G)

def generate_matrix(size, connection_chances, seed):
    #generate large adjacency matrices
    #because of the parsing method, this can be an invalid adj.mat - just random numbers
    #based on the examples we can try 2/3 chance of Y - but can try other numbers too, network theory
    np.random.seed(seed)
    gen_matrix = np.random.choice(["Y","N"],(size, size),p=connection_chances)
    
    def imageSwap(gen_matrix, size):
    #code from https://www.geeksforgeeks.org/python-program-for-mirror-of-matrix-across-diagonal/
    # traverse a matrix and swap
    # mat[i][j] with mat[j][i]
        for i in range(size):
            for j in range(i + 1):
                t = gen_matrix[i][j]
                gen_matrix[i][j] = gen_matrix[j][i]
                gen_matrix[j][i] = t
    
    imageSwap(gen_matrix, size)
    
    return(gen_matrix)

def find_and_show_potentials(Gmat):
    #find our man - largest amount of potential friends

    size = len(Gmat.nodes())
    largest_count = [-100, 0]
    neighbour_nodes = []
    potential_nodes = []

    for node in Gmat.nodes():
        potentials_curr = []
        for neigh in Gmat.neighbors(node):
            unwanted = list(Gmat.neighbors(node)) + [node]
            less_n = [ele for ele in Gmat.neighbors(neigh) if ele not in unwanted] #remove the node and neighbors
            potentials_curr = potentials_curr + less_n
        #print(set(potentials_curr))

        if len(set(potentials_curr)) > largest_count[1]:
            largest_count[0] = node
            largest_count[1] = len(set(potentials_curr))
            neighbour_nodes = list(Gmat.neighbors(node))
            potential_nodes = list(set(potentials_curr))
    
    #print a couple values
    print(f"The index of the searched node is {largest_count[0]} and the potential friend count is {largest_count[1]}")
    print(f"The index of its neighbors:{neighbour_nodes}")
    print(f"The index of its potential friends:{potential_nodes}")
    
    #The remaining nodes we get by removing the previous nodes from all
    others = [ele for ele in range(size) if ele not in neighbour_nodes and ele not in potential_nodes and ele != largest_count[0]]
 
    #color the node in col1, neighbors col2, potential friends in col3, others col4
    plt.figure(figsize=(12,7))
    pos = nx.kamada_kawai_layout(Gmat)  # positions for all nodes
    nx.draw_networkx_nodes(Gmat, pos, nodelist=[largest_count[0]], node_color="red")
    nx.draw_networkx_nodes(Gmat, pos, nodelist=neighbour_nodes, node_color="orange")
    nx.draw_networkx_nodes(Gmat, pos, nodelist=potential_nodes, node_color="green")
    nx.draw_networkx_nodes(Gmat, pos, nodelist=others, node_color="blue")

    nx.draw_networkx_edges(Gmat, pos, width=1.0)

    labels = {}
    for i in range(size):
        labels[i] = i    

    nx.draw_networkx_labels(Gmat, pos, labels, font_color="whitesmoke")

Red: the node we looked for, orange: its neigbors, green: potential friends, blue: at least three edge separation
